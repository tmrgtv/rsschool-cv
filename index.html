Take a Ten Minutes Walk

package kata

func IsValidWalk(walk []rune) bool {
  if len(walk) != 10 {
    return false
  }
  var x,y int
  for _,dir := range walk {
    switch dir {
      case 'n':
        x += 1
      case 's':
        x -= 1
      case 'e':
        y += 1
      case 'w':
        y -= 1
    }
  }

  if x == 0 && y == 0 {
    return true
  }

  return false
}
